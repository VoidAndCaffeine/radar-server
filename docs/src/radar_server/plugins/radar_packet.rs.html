6/// An Hdf5Object can be stored and retrieved from an HDF5 file.
7pub trait Hdf5Object{
8    /// Stores the object in the specified file.
9    ///
10    /// The object is stored in a group corresponding to its epoch time,
11    /// and a subgroup corresponding to its subsecond millisecond time.
12    fn to_hdf5(&self, file: &File) -> hdf5_metno::Result<()>;
13
14    /// Retrieves an object from the specified file and group.
15    ///
16    /// Only retrieves an object from a group or fails, does not search for a specific object.
17    fn from_hdf5(group: &Group, time: SystemTime) -> hdf5_metno::Result<Self> where Self: Sized;
18}

91/// Implementation of HDF5Object for ComPacketFloat.
92impl Hdf5Object for ComPacketFloat {
93    fn to_hdf5(&self, file: &File) -> hdf5_metno::Result<()> {
94        let packet_time = self.time.duration_since(SystemTime::UNIX_EPOCH)
95            .expect("Time went backwards");
96        let packet_secs = packet_time.as_secs().to_string();
97        let packet_ns = packet_time.subsec_nanos().to_string();
98
99        let super_group = file.create_group(packet_secs.as_str())?;
100        let group = super_group.create_group(packet_ns.as_str())?;
101
102        let attr = group.new_attr::<u8>().create("Identity")?;
103        let s = serde_json::to_string(&self.id).expect("Unable to serialize to json");
104        attr.write(s.as_bytes())?;
105        let attr = group.new_attr::<State>().create("State")?;
106        attr.write_scalar(&self.state)?;
107
108        let data_ds = group.new_dataset::<f64>()
109            .shape(self.data.len())
110            .create("Data")?;
111        data_ds.write(&self.data)?;
112        file.flush()?;
113        Ok(())
114    }
115    
116    fn from_hdf5(group: &Group, time:SystemTime) -> hdf5_metno::Result<Self> {
117        let id_attr = group.attr("Identity")?;
118        let id_bytes = id_attr.read_raw::<u8>()?;
119        let id_json = String::from_utf8(id_bytes).unwrap();
120        let id = serde_json::from_str(id_json.as_str()).expect("Unable to deserialize from json");
121
122        let state_attr = group.attr("State")?;
123        let state = state_attr.read_scalar()?;
124
125        let data_ds = group.dataset("Data")?;
126        let data = data_ds.read_raw::<f64>()?;
127
128        Ok(ComPacketFloat{id,time,state,data})
129    }
130}
